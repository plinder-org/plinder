51def explicit_H_remover(mol: Mol, remove_hydrogens: list[int]) -> Mol: 

52 """removes all H atoms in the list and all bonds to those hydrogens""" 

53 res = Chem.RWMol(mol) 

54 res.BeginBatchEdit() 

55 for aid in remove_hydrogens: 

56 neighbors = res.GetAtomWithIdx(aid).GetNeighbors() 

57 for neighbor in neighbors: 

58 res.RemoveBond(aid, neighbor.GetIdx()) 

59 res.RemoveAtom(aid) 

60 res.CommitBatchEdit() 

61 return res

64def fix_valency_issues(mol: Mol) -> Mol: 

65 """Fix valency issues with rdkit mol and return sanitized.

67 Removed hydrogens if there is an issue with their valence!

70 # NOT: Explicit valence for atom # X C, 5, is greater than permitted

179def remove_unmatched_atoms(mol: Chem.Mol, match: NDArray) -> Chem.Mol: 

180 """Remove atoms in mol whose indices are not in match. 

181 Parameters 

182 ---------- 

183 mol : Chem.Mol 

184 the mol to be modified 

185 match : NDArray 

186 indices that are matches and should not be removed 

187 Returns 

188 ------- 

189 Chem.Mol 

190 the mol with unmatched atoms removed 

191 """ 

192 res = Chem.RWMol(mol)

204 except:

210def get_matched_template(template: Chem.Mol, mol: Chem.Mol) -> Chem.Mol: 

211 """ 

212 Perform MCS matching between a (subject) mol and a template; and return the matched template with 

213 the bond orders of the template. Used to assign bond orders in 

214 `safe_mol_from_pdb_assign_bond_orders`. Known limitation: if the template has 

215 double/triple bonds and the mol doesn't (because it's read from PDB), this leads to 

216 removing all atoms that don't match, incl. the ones bound via e.g., a double bond. This is 

217 only a problem if we need to use this fallback option because previous attempts in 

218 `safe_mol_from_pdb_assign_bond_orders` have failed. 

219 Returns 

220 ------- 

221 Chem.Mol 

222 the matching template with bond orders from template 

223 """ 

224 # set all bonds to unspecified to help with the match 

225 match_mol = copy.deepcopy(mol) 

226 [b.SetBondType(Chem.BondType.UNSPECIFIED) for b in match_mol.GetBonds()] 

227 

228 mcs = FindMCS( 

229 [match_mol, template], 

230 completeRingsOnly=False, 

231 ringMatchesRingOnly=False, 

232 timeout=10, 

233 ) 

234 patt = Chem.MolFromSmarts(mcs.smartsString) 

235 atom_map_template = np.array(template.GetSubstructMatch(patt)) 

236 # remove all atoms from the ref that are not in the MCS --> use this as template for 

237 # bond orders 

238 matched_template_mol = remove_unmatched_atoms(template, atom_map_template) 

239 return matched_template_mol

242def remove_unmatched_atoms_and_bonds( 

243 mol: Chem.Mol, matched_atoms: NDArray, matched_bonds: NDArray 

244) -> Chem.Mol: 

245 """Remove atoms and bonds in mol whose indices are not in match. 

246 Parameters 

247 ---------- 

248 mol : Chem.Mol 

249 the mol to be modified 

250 match_bonds : NDArray 

251 indices that are matches and should not be removed 

252 matched_bonds : NDArray 

253 indices that are matches and should not be removed 

254 Returns 

255 ------- 

256 Chem.Mol 

257 the mol with unmatched atoms removed 

258 """ 

259 res = Chem.RWMol(mol) 

260 bonds_to_remove = [ 

261 (b.GetBeginAtomIdx(), b.GetEndAtomIdx()) 

262 for b in mol.GetBonds() 

263 if b.GetIdx() not in matched_bonds 

264 ] 

265 atoms_to_remove = [ 

266 a.GetIdx() for a in mol.GetAtoms() if a.GetIdx() not in matched_atoms 

267 ] 

268 res.BeginBatchEdit() 

269 for atom_idx in atoms_to_remove: 

270 neighbors = res.GetAtomWithIdx(atom_idx).GetNeighbors() 

271 for neighbor in neighbors: 

272 bonds_to_remove.append((atom_idx, neighbor.GetIdx())) 

273 # if atom to be removed is neighbour to double bond - set that stereo bond to undefined 

274 [ 

275 nb.SetStereo(Chem.rdchem.BondStereo.STEREONONE) 

276 for nb in neighbor.GetBonds() 

277 if nb.GetBondType() == Chem.rdchem.BondType.DOUBLE 

278 ] 

279 res.RemoveAtom(atom_idx) 

280 # now remove all unmatched bonds! 

281 for bond_idx1, bond_idx2 in bonds_to_remove: 

282 res.RemoveBond(bond_idx1, bond_idx2) 

283 

284 res.CommitBatchEdit() 

285 res = Chem.Mol(res) 

286 try: 

287 Chem.SanitizeMol(res) 

288 except: 

289 pass 

290 [a.SetNumRadicalElectrons(0) for a in res.GetAtoms()] 

291 return res
