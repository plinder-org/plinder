17class PlinderDataset(Dataset): # type: ignore 

18 """ 

19 Creates a dataset from plinder systems 

20 

21 Parameters 

22 ---------- 

23 df : pd.DataFrame | None 

24 the split to use 

25 split : str 

26 the split to sample from 

27 file_with_system_ids : str | Path 

28 path to a file containing a list of system ids (default: full index) 

29 store_file_path : bool, default=True 

30 if True, include the file path of the source structures in the dataset 

31 num_alternative_structures : int, default=0 

32 if available, load up to this number of alternative structures (apo and pred) 

33 """

35 def __init__( 

36 self, 

37 df: pd.DataFrame | None = None, 

38 split: str = "train", 

39 split_parquet_path: str | Path | None = None, 

40 store_file_path: bool = True, 

41 num_alternative_structures: int = 0, 

42 file_paths_only: bool = False, 

43 ): 

44 if df is None: 

45 if split_parquet_path is None: 

46 df = get_split() 

47 else: 

48 df = pd.read_parquet(split_parquet_path) 

49 self._system_ids = df.loc[df["split"] == split, "system_id"].to_list() 

50 self._num_examples = len(self._system_ids) 

51 self._store_file_path = store_file_path or file_paths_only 

52 self._links = None 

53 if num_alternative_structures > 0: 

54 self._links = query_links().groupby("reference_system_id") 

55 self.num_alternative_structures = num_alternative_structures 

56 self._file_paths_only = file_paths_only 

57 

58 def __len__(self) -> int: 

59 return self._num_examples 

60 

61 def __getitem__( 

62 self, index: int 

63 ) -> dict[str, int | str | pd.DataFrame | dict[str, str | pd.DataFrame]]: 

64 if not 0 <= index < self._num_examples: 

65 raise IndexError(index) 

66 

67 s = system.PlinderSystem(system_id=self._system_ids[index]) 

68 

69 if self._file_paths_only: 

70 # avoid loading structure if not needed 

71 structure_df = None 

72 else: 

73 structure_df = fo.bp_to_df(fo.read_any(s.system_cif)) 

74 

75 item: Dict[str, Any] = { 

76 "id": index, 

77 "system_id": s.system_id, 

78 "df": structure_df, 

79 "alternative_structures": {}, 

80 } 

81 if self._store_file_path: 

82 item["path"] = s.system_cif 

83 

84 if self._links is not None: 

85 alternatives: dict[str, pd.DataFrame | str | None] = {} 

86 try: 

87 links = self._links.get_group(s.system_id) 

88 if not links.empty: 

89 alts = links.groupby("kind").head(self.num_alternative_structures) 

90 # TODO: make better stagger between the kinds! 

91 count = 0 

92 for kind, link_id in zip(alts["kind"], alts["id"]): 

93 structure = s.get_linked_structure( 

94 link_kind=str(kind), link_id=link_id 

95 ) 

96 df_key = f"{kind}_{link_id}_df" 

97 if self._file_paths_only: 

98 alternatives[df_key] = None 

99 else: 

100 alternatives[df_key] = fo.bp_to_df(fo.read_any(structure)) 

101 if self._store_file_path: 

102 alternatives[f"{kind}_{link_id}_path"] = structure 

103 count += 1 

104 # if we have enough alternatives, return them 

105 if count >= self.num_alternative_structures: 

106 item["alternative_structures"] = alternatives 

107 return item 

108 except KeyError: 

109 pass 

110 item["alternative_structures"] = alternatives 

111 return item

114def get_model_input_files( 

115 split_df: pd.DataFrame, 

116 split: Literal["train", "val", "test"] = "train", 

117 max_num_sample: int = 10, 

118 num_alternative_structures: int = 1, 

119) -> List[Tuple[Path, str, List[Any]]]: 

120 model_inputs = [] 

121 

122 smiles_in_df = True 

123 if "ligand_rdkit_canonical_smiles" not in split_df.columns: 

124 smiles_in_df = False 

125 from rdkit import Chem 

126 

127 dataset = PlinderDataset( 

128 df=split_df, 

129 split=split, 

130 num_alternative_structures=num_alternative_structures, 

131 file_paths_only=True, 

132 ) 

133 

134 count = 0 

135 for data in dataset: 

136 system_dir = Path(data["path"]).parent 

137 protein_fasta_filepath = system_dir / "sequences.fasta" 

138 

139 if num_alternative_structures: 

140 alt_structure_filepaths = [ 

141 val 

142 for key, val in data["alternative_structures"].items() 

143 if key.endswith("_path") 

144 ] 

145 else: 

146 alt_structure_filepaths = [] 

147 

148 if smiles_in_df: 

149 smiles_array = split_df.loc[ 

150 split_df["system_id"] == data["system_id"], 

151 "ligand_rdkit_canonical_smiles", 

152 ].values 

153 else: 

154 smiles_array = [ 

155 Chem.MolToSmiles(next(Chem.SDMolSupplier(ligand_sdf))) 

156 for ligand_sdf in (system_dir / "ligand_files/").glob("*sdf") 

157 ] 

158 # in case there's more than one! 

159 smiles = ".".join(smiles_array) 

160 model_inputs.append((protein_fasta_filepath, smiles, alt_structure_filepaths)) 

161 count += 1 

162 if count >= max_num_sample: 

163 break 

164 return model_inputs

169def get_diversity_samples( 

170 split_df: pd.DataFrame, 

171 split: Literal["train", "val", "test"] = "train", 

172 cluster_tag: None | str = None, 

173 metric: Literal[ 

174 "pli_qcov", 

175 "pli_unique_qcov", 

176 "pocket_fident", 

177 "pocket_fident_qcov", 

178 "pocket_lddt", 

179 "pocket_lddt_qcov", 

180 "pocket_qcov", 

181 "protein_fident_max", 

182 "protein_fident_qcov_max", 

183 "protein_fident_qcov_weighted_max", 

184 "protein_fident_qcov_weighted_sum", 

185 "protein_fident_weighted_max", 

186 "protein_fident_weighted_sum", 

187 "protein_lddt_max", 

188 "protein_lddt_qcov_max", 

189 "protein_lddt_qcov_weighted_max", 

190 "protein_lddt_qcov_weighted_sum", 

191 "protein_lddt_weighted_max", 

192 "protein_lddt_weighted_sum", 

193 "protein_qcov_weighted_sum", 

194 "protein_seqsim_max", 

195 "protein_seqsim_qcov_max", 

196 "protein_seqsim_qcov_weighted_max", 

197 "protein_seqsim_qcov_weighted_sum", 

198 "protein_seqsim_weighted_max", 

199 "protein_seqsim_weighted_sum", 

200 ] = "pli_qcov", 

201 directed: bool = True, 

202 threshold: int = 100, 

203 cluster_type: Literal["component", "communities"] = "component", 

204) -> pd.DataFrame: 

205 from torch.utils.data import WeightedRandomSampler 

206 

207 if not cluster_tag: 

208 direction_tag = "strong" if directed else "weak" 

209 cluster_tag = f"{metric}__{threshold}__{direction_tag}__{cluster_type}" 

210 

211 split_df = split_df[split_df.split == split] 

212 cluster_counts = split_df[cluster_tag].value_counts().rename("cluster_count") 

213 split_df = split_df.merge(cluster_counts, left_on=cluster_tag, right_index=True) 

214 split_df.reset_index(inplace=True) 

215 cluster_weights = 1.0 / split_df.cluster_count.values 

216 sampler = WeightedRandomSampler( 

217 weights=cluster_weights, num_samples=len(cluster_weights) 

218 ) 

219 sampler_index = [i for i in sampler] 

220 return split_df.loc[tuple(sampler_index), ("system_id", "split")].drop_duplicates()
